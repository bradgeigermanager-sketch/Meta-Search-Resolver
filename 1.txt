# 3. Query Builder (Generalized)

```python
def build_queries(name=None, username=None, email=None, domain=None):
    q = []

    if name:
        q.append(f'"{name}"')
        q.append(f'"{name}" profile')
        q.append(f'"{name}" developer')
        q.append(f'"{name}" github')
        q.append(f'"{name}" linkedin')

    if username:
        q.append(username)
        q.append(f'"{username}"')
        q.append(f'{username} profile')

    if email:
        q.append(f'"{email}"')
        q.append(f'"{email}" github')
        q.append(f'"{email}" stackoverflow')

    if domain:
        q.append(f'site:{domain}')
        q.append(f'"{domain}" admin')

    return q
```

# 6. Meta‑Search URL Constructor

```python
def build_all_urls(queries):
    urls = []

    for q in queries:
        # Search engines
        for name, func in SEARCH_ENGINES.items():
            urls.append({"source": name, "query": q, "url": func(q)})

        # Site-specific
        urls.extend({"source": "github", "query": q, "url": u} for u in github_urls(q))
        urls.extend({"source": "gitlab", "query": q, "url": u} for u in gitlab_urls(q))
        urls.extend({"source": "stackoverflow", "query": q, "url": u} for u in stackoverflow_urls(q))
        urls.extend({"source": "reddit", "query": q, "url": u} for u in reddit_urls(q))
        urls.extend({"source": "npm", "query": q, "url": u} for u in npm_urls(q))
        urls.extend({"source": "pypi", "query": q, "url": u} for u in pypi_urls(q))
        urls.extend({"source": "dockerhub", "query": q, "url": u} for u in dockerhub_urls(q))

    return urls
```

# 7. Async Fetch Layer (Universal)

async def fetch_all(urls):
    async with aiohttp.ClientSession() as session:
        tasks = [fetch(session, u["url"]) for u in urls]
        return await asyncio.gather(*tasks)

def score_identity(username, signals):
    score = 0

    if signals.get("exact_match"):
        score += 100
    if signals.get("name_match"):
        score += 50
    if signals.get("cross_platform"):
        score += 80
    if signals.get("email_match"):
        score += 120
    if signals.get("activity"):
        score += 40

    return score

# 10. Orchestrator (Meta‑Resolver)

```python
async def resolve_identity(name=None, username=None, email=None, domain=None):
    queries = build_queries(name, username, email, domain)
    urls = build_all_urls(queries)
    html_pages = await fetch_all(urls)

    candidates = {}

    for html, meta in zip(html_pages, urls):
        extractor = EXTRACTORS.get(meta["source"])
        if extractor:
            found = extractor(html)
            for f in found:
                candidates.setdefault(f, []).append(meta["source"])

    # scoring
    results = []
    for user, sources in candidates.items():
        score = score_identity(user, {"cross_platform": len(sources) > 1})
        results.append({"username": user, "score": score, "sources": sources})

    return sorted(results, key=lambda x: x["score"], reverse=True)
```
