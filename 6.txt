# 🐦 Twitter / X Plugin

```python
# sites/twitter.py
import re
from urllib.parse import quote_plus
from .base import SitePlugin

class TwitterPlugin(SitePlugin):
    name = "twitter"

    def build_urls(self, query: str) -> list[str]:
        q = quote_plus(query)
        return [
            f"https://twitter.com/search?q={q}&src=typed_query",
            f"https://nitter.net/search?f=users&q={q}",
            f"https://www.google.com/search?q=site%3Atwitter.com+{q}",
        ]

    def extract(self, html: str) -> list[dict]:
        results = []
        profile_re = r"twitter\.com/([A-Za-z0-9_]{3,20})"
        for username in set(re.findall(profile_re, html)):
            results.append({
                "username": username,
                "profile_url": f"https://twitter.com/{username}",
                "extra": {"type": "profile"}
            })
        return results
```

# 📘 Facebook (Public Pages Only)

```python
# sites/facebook.py
import re
from urllib.parse import quote_plus
from .base import SitePlugin

class FacebookPlugin(SitePlugin):
    name = "facebook"

    def build_urls(self, query: str) -> list[str]:
        q = quote_plus(query)
        return [
            f"https://www.google.com/search?q=site%3Afacebook.com+{q}",
            f"https://www.facebook.com/public/{q}",
        ]

    def extract(self, html: str) -> list[dict]:
        results = []
        profile_re = r"facebook\.com/([A-Za-z0-9.\-]+)/?"
        for username in set(re.findall(profile_re, html)):
            if username.lower() in {"pages", "public", "groups"}:
                continue
            results.append({
                "username": username,
                "profile_url": f"https://www.facebook.com/{username}",
                "extra": {"type": "profile"}
            })
        return results
```

```python
# sites/youtube.py
import re
from urllib.parse import quote_plus
from .base import SitePlugin

class YouTubePlugin(SitePlugin):
    name = "youtube"

    def build_urls(self, query: str) -> list[str]:
        q = quote_plus(query)
        return [
            f"https://www.youtube.com/results?search_query={q}",
            f"https://www.google.com/search?q=site%3Ayoutube.com+{q}",
        ]

    def extract(self, html: str) -> list[dict]:
        results = []
        channel_re = r"youtube\.com/channel/([A-Za-z0-9_-]+)"
        handle_re = r"youtube\.com/@([A-Za-z0-9_.-]+)"

        for cid in set(re.findall(channel_re, html)):
            results.append({
                "username": cid,
                "profile_url": f"https://www.youtube.com/channel/{cid}",
                "extra": {"type": "channel_id"}
            })

        for handle in set(re.findall(handle_re, html)):
            results.append({
                "username": handle,
                "profile_url": f"https://www.youtube.com/@{handle}",
                "extra": {"type": "handle"}
            })

        return results
```

```python
import re
from urllib.parse import quote_plus
from .base import SitePlugin

class MastodonPlugin(SitePlugin):
    name = "mastodon"

    def build_urls(self, query: str) -> list[str]:
        q = quote_plus(query)
        return [
            f"https://mastodon.social/search?q={q}",
            f"https://www.google.com/search?q={q}+mastodon",
        ]

    def extract(self, html: str) -> list[dict]:
        results = []
        acct_re = r"@([A-Za-z0-9_]+)@([A-Za-z0-9\.-]+)"
        for user, server in set(re.findall(acct_re, html)):
            results.append({
                "username": f"{user}@{server}",
                "profile_url": f"https://{server}/@{user}",
                "extra": {"server": server}
            })
        return results
```
