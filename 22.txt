### Unique‑ID extraction plugin pack

Below are upgraded versions of each **site plugin** with **unique‑ID extraction** wired in.  
Each plugin now fills `identifiers` on the returned account objects.

Assumes:

- `Account` has `identifiers: Dict[str, str]`
- You still wrap these dicts into `Account` later (as in your existing pipeline).

---

#### GitHub unique‑ID extraction

```python
# plugins/sites/github.py
import re
from urllib.parse import quote_plus
from plugins.base import SitePlugin

class GitHubPlugin(SitePlugin):
    name = "github"

    def build_urls(self, query: str) -> list[str]:
        q = quote_plus(query)
        return [
            f"https://github.com/search?q={q}&type=users",
            f"https://www.google.com/search?q=site%3Agithub.com+{q}"
        ]

    def extract(self, html: str) -> list[dict]:
        results = []
        user_re = r"github\.com/([A-Za-z0-9_-]+)"
        id_re = r'data-hovercard-user-id="(\d+)"'

        usernames = set(re.findall(user_re, html))
        ids = re.findall(id_re, html)

        # naive pairing: same order; you can improve with better parsing
        id_iter = iter(ids)

        for username in usernames:
            user_id = next(id_iter, None)
            identifiers = {}
            if user_id:
                identifiers["user_id"] = user_id

            results.append({
                "username": username,
                "profile_url": f"https://github.com/{username}",
                "display_name": None,
                "bio": None,
                "location": None,
                "email": None,
                "avatar_hash": None,
                "links": [],
                "identifiers": identifiers,
                "extra": {"type": "profile"}
            })
        return results
```

import re
from urllib.parse import quote_plus

                "bio": None,
                "location": None,
                "email": None,
                "avatar_hash": None,
                "links": [],
                "identifiers": identifiers,
                "extra": {"user_id": uid}
            })
        return results

import re
from urllib.parse import quote_plus
from plugins.base import SitePlugin

class FacebookPlugin(SitePlugin):
    name = "facebook"

    def build_urls(self, query: str) -> list[str]:
        q = quote_plus(query)

            f"https://www.google.com/search?q=site%3Afacebook.com+{q}",

        ]

    def extract(self, html: str) -> list[dict]:

            profile_id = next(id_iter, None)
            identifiers = {}
            if profile_id:
                identifiers["profile_id"] = profile_id

            results.append({
                "username": username,
                "profile_url": f"https://www.facebook.com/{username}",
                "display_name": None,
                "bio": None,
                "location": None,
                "email": None,
                "avatar_hash": None,
                "links": [],
                "identifiers": identifiers,

import re
from urllib.parse import quote_plus
from plugins.base import SitePlugin

class YouTubePlugin(SitePlugin):
    name = "youtube"

    def build_urls(self, query: str) -> list[str]:
        q = quote_plus(query)
        return [
            f"https://www.youtube.com/results?search_query={q}",
            f"https://www.google.com/search?q=site%3Ayoutube.com+{q}"
        ]

    def extract(self, html: str) -> list[dict]:
        results = []
        channel_re = r"youtube\.com/channel/([A-Za-z0-9_-]+)"
        handle_re = r"youtube\.com/@([A-Za-z0-9_.-]+)"
        chanid_re = r'"channelId":"([^"]+)"'

        channel_ids = set(re.findall(channel_re, html))
        handles = set(re.findall(handle_re, html))
        chan_ids = re.findall(chanid_re, html)
        chan_iter = iter(chan_ids)

        for cid in channel_ids:
            channel_id = next(chan_iter, None)
            identifiers = {}
            if channel_id:
                identifiers["channel_id"] = channel_id

            results.append({
                "username": cid,
                "profile_url": f"https://www.youtube.com/channel/{cid}",
                "display_name": None,
                "bio": None,
                "location": None,
                "email": None,
                "avatar_hash": None,
                "links": [],
                "identifiers": identifiers,
                "extra": {"type": "channel_id"}
            })

        for handle in handles:
            channel_id = next(chan_iter, None)
            identifiers = {}
            if channel_id:
                identifiers["channel_id"] = channel_id

            results.append({
                "username": handle,
                "profile_url": f"https://www.youtube.com/@{handle}",
                "display_name": None,
                "bio": None,
                "location": None,
                "email": None,
                "avatar_hash": None,
                "links": [],
                "identifiers": identifiers,
                "extra": {"type": "handle"}
            })

        return results
