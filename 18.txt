```python
from typing import Dict, Any

@dataclass
class Node:
    id: str
    type: str
    label: str
    attributes: Dict[str, Any] = field(default_factory=dict)

@dataclass
class Edge:
    id: str
    source: str
    target: str
    type: str
    weight: float
    plugin: str
    evidence: Dict[str, Any] = field(default_factory=dict)
```

# 🧠 2. Graph Builder (Collects Nodes + Edges From Plugins)

```python
# core/graph_builder.py
import uuid
from core.graph_model import Node, Edge

class IdentityGraph:
    def __init__(self):
        self.nodes = {}
        self.edges = []

    def add_node(self, node: Node):
        self.nodes[node.id] = node

    def add_edge(self, edge: Edge):
        self.edges.append(edge)

    def to_json(self):
        return {
            "nodes": [vars(n) for n in self.nodes.values()],
            "edges": [vars(e) for e in self.edges]
        }
```

```python
import re

def extract_artifacts_from_account(account):
    artifacts = []

    # email

        artifacts.append({
            "id": f"artifact:email:{account['email']}",
```

# 🧩 5. Graph Plugins Generate Edges

```python
# core/run_graph_plugins.py
def build_graph_edges(graph_plugins, accounts, artifacts):
    edges = []

    for plugin in graph_plugins:
        plugin_edges = plugin.build_edges(accounts, artifacts)
        for e in plugin_edges:
            e.id = e.id or str(uuid.uuid4())
            edges.append(e)

    return edges
```

# 🧠 6. Identity Clustering (Connected Components)

```python
# core/identity_cluster.py
import networkx as nx

def cluster_identities(edges):
    G = nx.Graph()

    for e in edges:
        G.add_edge(e.source, e.target, weight=e.weight)

    clusters = []
    for comp in nx.connected_components(G):
        clusters.append(list(comp))

    return clusters
```

```python
# core/build_identity_nodes.py
def build_identity_nodes(graph, clusters):
    for idx, cluster in enumerate(clusters):
        identity_id = f"identity:id_{idx+1}"

        graph.add_node(Node(
            id=identity_id,
            type="identity",
            label=f"Identity #{idx+1}",
            attributes={"cluster_size": len(cluster)}
        ))

        for node_id in cluster:
            graph.add_edge(Edge(
                id=str(uuid.uuid4()),
                source=identity_id,
                target=node_id,
                type="IDENTITY_HAS_ACCOUNT",
                weight=1.0,
                plugin="identity_resolver",
                evidence={}
            ))
```

# 🧩 8. Full Pipeline to Build the JSON Graph

```python
# core/build_identity_graph.py
from core.graph_builder import IdentityGraph
from core.build_accounts import build_account_nodes
from core.build_artifacts import extract_artifacts_from_account
from core.run_graph_plugins import build_graph_edges
from core.identity_cluster import cluster_identities
from core.build_identity_nodes import build_identity_nodes

def build_identity_graph(query, site_plugins, graph_plugins):
    graph = IdentityGraph()

    # 1. Collect accounts
    accounts = build_account_nodes(site_plugins, query)

    # 2. Add account nodes
    for a in accounts:
        graph.add_node(Node(
            id=a["node_id"],
            type="account",
            label=f"{a['plugin']}:{a['username']}",
            attributes=a
        ))

    # 3. Collect artifacts
    artifacts = []
    for a in accounts:
        artifacts.extend(extract_artifacts_from_account(a))

    # 4. Add artifact nodes
    for art in artifacts:
        graph.add_node(Node(**art))

    # 5. Build edges from graph plugins
    edges = build_graph_edges(graph_plugins, accounts, artifacts)
    for e in edges:
        graph.add_edge(e)

    # 6. Cluster identities
    clusters = cluster_identities(edges)

    # 7. Add identity nodes + edges
    build_identity_nodes(graph, clusters)

    return graph.to_json()
```
