## 0. Base interfaces and shared types

```python
# plugins/base.py
from abc import ABC, abstractmethod
from dataclasses import dataclass, field
from typing import Any, Dict, List

# ---- Core account model ----
@dataclass
class Account:
    node_id: str
    source: str
    username: str
    profile_url: str | None = None
    display_name: str | None = None
    bio: str | None = None
    location: str | None = None
    email: str | None = None
    avatar_hash: str | None = None
    links: List[str] = field(default_factory=list)
    extras: Dict[str, Any] = field(default_factory=dict)


# ---- Site plugins ----
class SitePlugin(ABC):
    name: str

    @abstractmethod
    def build_urls(self, query: str) -> list[str]:
        ...

    @abstractmethod
    def extract(self, html: str) -> list[dict]:
        """
        Returns list of dicts with at least:
        {
          "username": str,
          "profile_url": str | None,
          "display_name": str | None,
          "bio": str | None,
          "location": str | None,
          "email": str | None,
          "avatar_hash": str | None,
          "links": list[str],
          "extra": dict
        }
        """
        ...


# ---- Similarity plugins ----
class SimilarityPlugin(ABC):
    name: str

    @abstractmethod
    def compare(self, a: Account, b: Account) -> dict:
        """
        Returns:
        {
          "score": float (0–1),
          "signals": dict
        }
        """
        ...


# ---- Graph plugins ----
@dataclass
class Edge:
    id: str | None
    src: str
    dst: str
    type: str
    weight: float
    plugin: str
    evidence: Dict[str, Any] = field(default_factory=dict)


class GraphPlugin(ABC):
    name: str

    @abstractmethod
    def build_edges(self, accounts: list[Account], artifacts: list[dict]) -> list[Edge]:
        ...
```

## 1. Site plugins (social + dev/info sources)

All of these follow the same pattern: build URLs + regex‑based extraction. You can later swap in API calls or more robust parsers.

```python
# plugins/sites/github.py
import re
from urllib.parse import quote_plus
from plugins.base import SitePlugin

class GitHubPlugin(SitePlugin):
    name = "github"

    def build_urls(self, query: str) -> list[str]:
        q = quote_plus(query)
        return [
            f"https://github.com/search?q={q}&type=users",
            f"https://www.google.com/search?q=site%3Agithub.com+{q}"
        ]

    def extract(self, html: str) -> list[dict]:
        results = []
        user_re = r"github\.com/([A-Za-z0-9_-]+)"
        for username in set(re.findall(user_re, html)):
            results.append({
                "username": username,
                "profile_url": f"https://github.com/{username}",
                "display_name": None,
                "bio": None,
                "location": None,
                "email": None,
                "avatar_hash": None,
                "links": [],
                "extra": {"type": "profile"}
            })
        return results
```

from rapidfuzz import fuzz

class UsernameSimilarityPlugin(SimilarityPlugin):

            "signals": {

# plugins/similarity/real_name_matching.py
from plugins.base import SimilarityPlugin, Account

def normalize_name(name: str) -> str:
    return name.lower().replace(".", "").strip()

class RealNameMatchingPlugin(SimilarityPlugin):
    name = "real_name_matching"

    def compare(self, a: Account, b: Account):
        if not a.display_name or not b.display_name:
            return {"score": 0.0, "signals": {}}
        na = normalize_name(a.display_name)
        nb = normalize_name(b.display_name)
        score = 1.0 if na == nb else 0.0
        return {
            "score": score,
            "signals": {
                "display_name_a": a.display_name,
                "display_name_b": b.display_name
            }
        }

from plugins.base import SimilarityPlugin, Account

class ExternalLinkReusePlugin(SimilarityPlugin):
    name = "external_link_reuse"

    def compare(self, a: Account, b: Account):
        set_a = set(a.links or [])
        set_b = set(b.links or [])
        if not set_a or not set_b:
            return {"score": 0.0, "signals": {}}
        overlap = set_a & set_b
        score = len(overlap) / max(len(set_a | set_b), 1)
        return {
            "score": score,
            "signals": {
                "overlap_links": list(overlap)
            }
        }

# plugins/similarity/bio_overlap.py
from plugins.base import SimilarityPlugin, Account

def tokenize(text: str) -> set[str]:
    return set(text.lower().split())

class BioOverlapPlugin(SimilarityPlugin):
    name = "bio_overlap"

    def compare(self, a: Account, b: Account):
        if not a.bio or not b.bio:
            return {"score": 0.0, "signals": {}}
        ta = tokenize(a.bio)
        tb = tokenize(b.bio)
        inter = ta & tb
        union = ta | tb
        score = len(inter) / max(len(union), 1)
        return {
            "score": score,
            "signals": {
                "overlap_terms": list(inter)
            }
        }

## 3. Graph plugins (edges for the mesh)

Each graph plugin uses one similarity signal (or direct data) to create `Edge` objects between accounts or between accounts and artifacts.

```python
# plugins/graph/username_graph.py
import uuid
from plugins.base import GraphPlugin, Edge, Account
from plugins.similarity.username_similarity import UsernameSimilarityPlugin

class UsernameGraphPlugin(GraphPlugin):
    name = "username_graph"

    def __init__(self):
        self.sim = UsernameSimilarityPlugin()

    def build_edges(self, accounts: list[Account], artifacts: list[dict]) -> list[Edge]:
        edges = []
        for i, a in enumerate(accounts):
            for j in range(i+1, len(accounts)):
                b = accounts[j]
                res = self.sim.compare(a, b)
                if res["score"] >= 0.7:
                    edges.append(Edge(
                        id=str(uuid.uuid4()),
                        src=a.node_id,
                        dst=b.node_id,
                        type="USERNAME_SIMILARITY",
                        weight=res["score"],
                        plugin=self.name,
                        evidence=res["signals"]
                    ))
        return edges
```

        self.photo_sim = PhotoReusePlugin()
        self.meme_sim = MemeReusePlugin()

    def build_edges(self, accounts: list[Account], artifacts: list[dict]) -> list[Edge]:
        edges = []
        for i, a in enumerate(accounts):

                    edges.append(Edge(
                        id=str(uuid.uuid4()),
                        src=a.node_id,
                        dst=b.node_id,
                        type="CONTENT_REUSE",
                        weight=score,
                        plugin=self.name,
                        evidence=evidence
                    ))
